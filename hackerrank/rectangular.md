#define ll long long

#define MIN(a, b) a < b ? a : b
#define MAX(a, b) a > b ? a : b

using namespace std;

int readline(char *str) {
    int i = 0;
    char ch;
    while((ch = getchar()) != '\n') {
        str[i++] = ch;
    }
    str[i] = '\0';
    return i;
}

int main(int argc, char *argv[]) {
    int t;
    scanf("%d", &t);

    ll int m = INT_MAX, n = INT_MAX;

    while(t--) {
        ll int a, b;
        cin >> a >> b;

        m = min(m, a);
        n = min(n, b);
    }

    cout << m * n << endl;

    return 0;
}
