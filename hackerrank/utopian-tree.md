using namespace std;

int readline(char *str) {
    int i = 0;
    char ch;
    while((ch = getchar()) != '\n') {
        str[i++] = ch;
    }
    str[i] = '\0';
    return i;
}

int main(int argc, char *argv[]) {
    int t;
    scanf("%d", &t);

    while(t--) {
        int n;
        ll int h = 1;

        cin >> n;
        for(int i = 1 ; i <= n ; i++) {
            if(i & 1) {
                h <<= 1;
            }
            else {
                h++;
            }
        }

    }
